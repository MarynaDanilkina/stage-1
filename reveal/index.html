Using npm:


							$ npm i -g npm

each method

Native forEach

							[1,2].forEach(function (item) {
								console.log(item);
							});
each lodash

							_.each([1,2], function (item) {
								console.log(item);
							});

map method

Native Map

							let newArr = [1,2,3].map(function (item) {
							  return item;
							})
Map lodash

							_.map([{id: 1}, {id: 2}, {id: 3}], function (item) {
							  return item.id;
							})

filter method

filter lodash

							_.filter([1,2,3,4,5], function (item) {
							  return item < 3;
							})
object filter

							_.filter([
								{id: 1, name: 'foo'}, {id:2, name: 'bar'}
								], function (item) {
							  	return item.name === 'foo';
							});

find method

find lodash

							_.find([
								{id: 1, name: 'foo'}, {id:2, name: 'bar'}
								], function (item){
								  return item.name === 'bar';
							});
short entry

							_.find([{id: 1, name: 'foo'}, {id:2, name: 'bar'}], {id: 1});

assign method


							let foo = { a: "a property" };
							let bar = { b: 4, c: "an other property" }
							let result = _.assign({ a: "an old property" }, foo, bar);
							// result => { a: 'a property', b: 4, c: 'an other property' }

sortedUniq method


							let sortedArray = [1, 1, 2, 3, 3, 3, 5, 8, 8];
							let result = _.sortedUniq(sortedArray);
							// -> [1, 2, 3, 5, 8]
